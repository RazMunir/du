Dynamic Dropdown Population

Examples of populating dropdowns using JavaScript

1. Populate from JavaScript Array

Selected value will appear here
// JavaScript array const languages = ["JavaScript", "Python", "Java", "C#", "PHP", "Ruby", "Swift"]; // Populate dropdown const dropdown = document.getElementById('arrayDropdown'); languages.forEach(lang => { const option = document.createElement('option'); option.value = lang.toLowerCase(); option.textContent = lang; dropdown.appendChild(option); }); // Event listener dropdown.addEventListener('change', function() { document.getElementById('arrayResult').textContent = this.value ? `You selected: ${this.value}` : 'Please select an option'; });

2. Populate from JavaScript Object

Selected value will appear here
// JavaScript object const frameworks = { "react": "React", "vue": "Vue.js", "angular": "Angular", "svelte": "Svelte", "ember": "Ember.js" }; // Populate dropdown const frameworkDropdown = document.getElementById('objectDropdown'); for (const [value, text] of Object.entries(frameworks)) { const option = document.createElement('option'); option.value = value; option.textContent = text; frameworkDropdown.appendChild(option); } // Event listener frameworkDropdown.addEventListener('change', function() { document.getElementById('objectResult').textContent = this.value ? `You selected: ${this.value} (${frameworks[this.value]})` : 'Please select an option'; });

3. Add Custom Options Dynamically

Selected value will appear here
// Add custom options document.getElementById('addOptionBtn').addEventListener('click', function() { const input = document.getElementById('newOptionInput'); const optionText = input.value.trim(); if (optionText) { const dropdown = document.getElementById('customDropdown'); const option = document.createElement('option'); option.value = optionText.toLowerCase().replace(/\s+/g, '_'); option.textContent = optionText; dropdown.appendChild(option); input.value = ''; input.focus(); } }); // Event listener document.getElementById('customDropdown').addEventListener('change', function() { document.getElementById('customResult').textContent = this.value ? `You selected: ${this.value}` : 'Please select an option'; });
